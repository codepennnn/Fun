SELECT *
FROM App_COMPLAINT_HELP_REGISTER
WHERE ID IN (
        SELECT MasterID
        FROM App_COMPLAINT_HELP_DETAILS
        WHERE FBACK_BY_VENDDATE >= DATEADD(MONTH, -3, CAST(GETDATE() AS DATE))
          AND DATENAME(WEEKDAY, FBACK_BY_VENDDATE) IN ('Saturday', 'Sunday')
     )
  AND COMPLAINT_DATE >= DATEADD(MONTH, -3, CAST(GETDATE() AS DATE))
  AND DATENAME(WEEKDAY, COMPLAINT_DATE) IN ('Saturday', 'Sunday')
ORDER BY COMPLAINT_DATE;
