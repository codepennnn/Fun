SELECT *
FROM App_COMPLAINT_HELP_DETAILS
WHERE COMPLAINT_DATE >= DATEADD(MONTH, -3, CAST(GETDATE() AS DATE))
  AND DATENAME(WEEKDAY, COMPLAINT_DATE) IN ('Saturday', 'Sunday');



  SELECT d.*, m.*
FROM App_COMPLAINT_HELP_DETAILS d
JOIN App_COMPLAINT_HELP_REGISTER m
    ON d.MasterID = m.ID
WHERE d.COMPLAINT_DATE >= DATEADD(MONTH, -3, CAST(GETDATE() AS DATE))
  AND DATENAME(WEEKDAY, d.COMPLAINT_DATE) IN ('Saturday', 'Sunday');
